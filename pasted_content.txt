      const excH=Math.max(0,saldoBruto-bancoMin);
      const excA=Math.max(0,saldoAnt-bancoMin);
      he50Min=Math.max(0,excH-excA);
    }
    const heDomMin=calc&&calc.cem?calc.totalMin:0;
    const valHE=(he50Min/60)*he50+(heDomMin/60)*he100;
    const valNot=calc?(calc.notMin/60)*(hn-hv):0;
    return{...dia,calc,saldoBruto,debitoMin,saldoMin,he50Min,heDomMin,valHE,valNot};
  });
};
